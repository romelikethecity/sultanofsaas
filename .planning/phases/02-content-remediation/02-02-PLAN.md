import sys; sys.path.insert(0, 'scripts')
from audit import check_writing
from build import TOOLS, NICHES, INDUSTRIES
errors = 0
# Check TOOLS data
for slug, tool_data in TOOLS.items():
    text = {'verdict_line': tool_data.get('verdict_line',''), 'best_for': tool_data.get('best_for',''), 'pros': tool_data.get('pros',[]), 'cons': tool_data.get('cons',[])}
    for f in check_writing(slug, text):
        if f.severity == 'error':
            print(f'TOOLS {slug}/{f.section}: {f.message}')
            errors += 1
# Check NICHES
for ns, n in NICHES.items():
    nt = {}
    if n.get('intro'): nt['intro'] = n['intro']
    if n.get('why_winners'): nt['why_winners'] = n['why_winners']
    if nt:
        for f in check_writing(f'niche:{ns}', nt):
            if f.severity == 'error':
                print(f'NICHE {ns}/{f.section}: {f.message}')
                errors += 1
# Check INDUSTRIES
for ins, ind in INDUSTRIES.items():
    it = {}
    if ind.get('intro'): it['intro'] = ind['intro']
    pt = {}
    for cs, p in ind.get('picks',{}).items():
        if isinstance(p, dict) and p.get('why'): pt[cs] = p['why']
    if pt: it['picks'] = pt
    if it:
        for f in check_writing(f'industry:{ins}', it):
            if f.severity == 'error':
                print(f'IND {ins}/{f.section}: {f.message}')
                errors += 1
print(f'Total TOOLS/NICHES/INDUSTRIES errors: {errors}')
sys.exit(1 if errors else 0)
